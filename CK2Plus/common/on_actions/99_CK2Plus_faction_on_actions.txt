# FROM is attacker
# ROOT is defender
on_war_ended_victory = {
	effect = {
		FROM = {
			add_character_modifier = {
				name = victory_timer
				duration = 1
				hidden = yes
			}
			pf_glory_plus2_effect = yes
		}
		ROOT = {
			pf_glory_minus2_effect = yes
		}
	}
}

# FROM is attacker
# ROOT is defender
on_war_ended_defeat = {
	effect = {
		FROM = {
			pf_glory_minus2_effect = yes
		}
		ROOT = {
			pf_glory_plus2_effect = yes
		}
	}
}

# FROM is attacker
# ROOT is defender
on_war_ended_victory = {
	effect = {
		FROM = {
			pf_glory_plus2_effect = yes
		}
		ROOT = {
			pf_glory_minus2_effect = yes
		}

		if = {
			limit = {
				OR = {
					using_cb = crusade
					using_cb = new_crusade
					using_cb = install_antiking
					using_cb = holy_war
					using_cb = religious_overthrow_ruler
					using_cb = italian_reconquest
					using_cb = reconquista_cb
				}
			}

			any_attacker = {
				pf_tradition_plus_4_effect = yes
			}
			any_defender = {
				pf_tradition_minus_4_effect = yes
			}
		}
	}
}

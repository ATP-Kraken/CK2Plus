add_cousin_opinion_effect = {
	any_close_relative = {
		limit = { is_aunt_uncle_of = ROOT }

		any_child = {
			limit = {
				is_cousin_of = ROOT # Redundant for safety
				NOT = {
					has_opinion_modifier = {
						who = ROOT
						modifier = opinion_cousin
					}
				}
			}

			opinion = {
				modifier = opinion_cousin
				who = ROOT
				years = 100
			}
			reverse_opinion = {
				modifier = opinion_cousin
				who = ROOT
				years = 100
			}
		}
	}
}

		}

			}
		}
	}
}

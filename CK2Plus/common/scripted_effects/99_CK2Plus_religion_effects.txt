indian_caste_fix_effect = {
	if = {
		limit = {
			OR = {
				religion = buddhist
				religion = hindu
				religion = jain
			}
			NOR = {
				trait = brahmin
				trait = kshatriya
				trait = vaishya
			}
		}

		if = {
			limit = {
				is_feudal = yes
			}
			add_trait = kshatriya
		}
		else_if = {
			limit = {
				is_feudal = no
				OR = {
					is_theocracy = yes
					is_priest = yes
				}
			}
			add_trait = brahmin
		}
		else = {
			add_trait = vaishya
		}
	}
}

decreased_crown_authority_effect = {
	holder_scope = {
		any_courtier_or_vassal = {
			limit = {
				NOT = {
					has_opinion_modifier = {
						who = PREV
						modifier = opinion_lowered_crown_authority
					}
				}
			}
			if = {
				limit = { is_voter = yes }

				opinion = {
					who = PREV
					modifier = opinion_lowered_crown_authority
					multiplier = 2
					years = 10
				}
			}
			else = {
				opinion = {
					who = PREV
					modifier = opinion_lowered_crown_authority
					years = 10
				}
			}
		}
	}
}

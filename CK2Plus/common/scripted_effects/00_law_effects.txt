# Council Power
decreased_council_power_effect = {
	holder_scope = {
		any_courtier_or_vassal = {
			limit = {
				OR = {
					is_voter = yes
					is_ruler = yes
				}
				NOT = {
					has_opinion_modifier = {
						name = opinion_decreased_council_power
						who = PREV
					}
				}
			}
			if = {
				limit = { is_voter = yes }

				opinion = {
					name = opinion_decreased_council_power
					who = PREV
					multiplier = 2
					years = 5
				}
			}
			else = {
				opinion = {
					name = opinion_decreased_council_power
					who = PREV
					years = 5
				}
			}
		}
	}
}

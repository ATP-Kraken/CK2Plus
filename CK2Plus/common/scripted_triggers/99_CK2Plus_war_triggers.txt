# Scopes from attacker
can_attack_patriarch_trigger = {
	trigger_if = {
		limit = {
			religion_group = christian
			has_pentarchy = yes

			PREV = {
				is_holy_site = ROOT # generally speaking, the holy sites are the pentarchies
				holder_scope = { religion = ROOT }
			}
		}

		higher_real_tier_than = DUKE
		NOT = { same_realm = FROM }
	}
}

# Scopes from attacker
can_attack_tributary_trigger = {
	trigger_if = {
		limit = {
			any_tributary = {
				OR = {
					character = FROM
					is_liege_or_above_of = FROM
				}
			}
		}

		OR = {
			trait = deceitful
			trait = arbitrary
		}
	}
}

# Scopes from attacker
can_attack_religion_head_trigger = {
	trigger_if = {
		limit = {
			rightful_religious_head = defender
			OR = {
				religion_group = christian
				religion_group = jewish_group
			}
		}

		OR = {
			trait = cynical
			trait = lunatic
			trait = excommunicated
			has_character_modifier = voice_of_satan
		}
	}
}

# Scopes from attacker for ease of use
can_use_cb_preamble = {
	NOT = { has_character_modifier = holy_truce_timer }

	# TODO: Move the below to ai_will_do for the relevant CBs
	trigger_if = {
		limit = { ai = yes }
		is_inaccessible_trigger = no
	}

	FROM = {
		NOT = { has_character_modifier = holy_truce_timer }

		trigger_if = {
			limit = {
				is_tributary = yes
				NOT = { pays_tribute_to = PREV }
			}

			suzerain = {
				NOR = {
					has_non_aggression_pact_with = PREVPREV
					is_allied_with = PREVPREV
				}
			}
		}
	}
}

# Current scope is potential attacker
# FROM is potential defender
can_use_holy_war_cb = {
	pacifist = no

	# Basic Requirements
	trigger_if = {
		limit = { religion_group = FROM }

		NOT = { religion = FROM }

		OR = {
			is_heresy_of = FROM
			is_parent_religion = FROM
			religion_group = indian_group
			religion_group = muslim
			religion_group = pagan_group
			religion = paulician
		}
	}

	# Same Realm
	trigger_if = {
		limit = {
			same_realm = FROM
			NOT = { religion_group = top_liege }
		}

		culture_group = tibetan_group

		OR = {
			religion = buddhist
			religion = bon_pagan
		}
	}
	trigger_else_if = {
		limit = { same_realm = FROM }
		religion_group = top_liege
	}

	# Specific Religion Requirements
	trigger_if = { # Pagans
		limit = { religion_group = pagan_group }

		OR = {
			is_reformed_religion = yes
			num_of_holy_sites >= 3
		}
	}
	trigger_else_if = { # Buddhist and Hindu
		limit = {
			OR = {
				religion = buddhist
				religion = hindu
			}

			FROM = { religion_group = indian_group }
		}

		FROM = {
			NOR = {
				religion = buddhist
				religion = hindu
				religion = jain
			}
		}
	}
	trigger_else_if = { # Cathar
		limit = { religion = cathar }
		has_global_flag = cathar_war
	}

	# Patricians
	trigger_if = {
		limit = { is_patrician = yes }
		independent = yes
	}
}

# Scopes from attacker
in_command_trigger = {
	trigger_if = {
		limit = {
			OR = {
				any_liege = { war = yes }
				war = yes
			}
		}

		in_command = yes
	}
	trigger_else = {
		hidden_trigger = { always = no }
	}
}

# Scopes from attacker
sufficient_treasury_for_war_trigger = {
	# Does not have a loan OR Does not have enough treasury to wage war
	check_variable = {
		which = loan_amount
		value < 50
	}

	trigger_if = {
		limit = { real_tier = BARON }
		wealth > 50
	}
	trigger_else_if = {
		limit = { real_tier = COUNT }
		wealth > 100
	}
	trigger_else_if = {
		limit = { real_tier = DUKE }
		wealth > 150
	}
	trigger_else_if = {
		limit = { real_tier = KING }
		wealth > 200
	}
	trigger_else = {
		wealth > 300
	}
}

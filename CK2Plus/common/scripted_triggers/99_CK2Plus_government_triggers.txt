imperial_government_type_trigger = {
	primary_title = {
		trigger_if = {
			limit = {
				has_game_started = no
				is_save_game = no
			}

			OR = {
				title = e_roman_empire
				title = e_byzantium
			}
		}
		trigger_else = {
			OR = {
				has_law = succ_byzantine_elective
				has_law = imperial_administration
				has_law = ze_administration_laws_2
			}
		}
	}
}

# Special Bogomilist Government
bogomilist_government_potential_trigger = {
	religion = bogomilist
	is_patrician = no

	NOR = {
		is_government_potential = roman_imperial_government
		is_government_potential = order_government
		is_government_potential = chinese_imperial_government
		is_government_potential = muslim_imperial_government
	}
}
bogomilist_government_allow_trigger = {}

# Imperial Muslim
muslim_imperial_government_potential_trigger = {
	trigger_if = {
		limit = { independent = yes }

		imperial_government_type_trigger = yes
		religion_group = muslim
		NOT = { religion = druze }
	}
	trigger_else = {
		top_liege = {
			imperial_government_type_trigger = yes
			religion_group = muslim
			NOT = { religion = druze }
		}
	}
}
muslim_imperial_government_allow_trigger = {}

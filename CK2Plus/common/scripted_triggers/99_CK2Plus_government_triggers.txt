imperial_government_type_trigger = {
	primary_title = {
		OR = {
			has_law = imperial_administration
			has_law = ze_administration_laws_2
		}
	}
}

# Imperial Muslim
muslim_imperial_government_potential_trigger = {
	muslim_government_potential_trigger = yes

	trigger_if = {
		limit = { independent = yes }
		imperial_government_type_trigger = yes
	}
	trigger_else = {
		top_liege = { muslim_imperial_government_potential_trigger = yes }
	}
}
muslim_imperial_government_allow_trigger = {}

# Special Bogomilist Government
bogomilist_government_potential_trigger = {
	religion = bogomilist

	chinese_imperial_government_potential_trigger = no
	order_government_potential_trigger = no
}
bogomilist_government_allow_trigger = {
	trigger_if = {
		limit = { has_game_started = yes }
		OR = {
			is_feudal = yes
			is_theocracy = yes
		}
	}
}

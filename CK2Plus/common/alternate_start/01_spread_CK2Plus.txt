culture_group_spread = {
	mult_modifier = {
		factor = 1.5 # Boost groups with many individual cultures somewhat
		OR = {
			culture_group = amazingh
			culture_group = central_germanic
			culture_group = finno_ugric
			culture_group = italian_group
			culture_group = south_slavic
		}
	}
	mult_modifier = {
		factor = 3 # Boost groups with very many individual cultures even more
		OR = {
			culture_group = amazingh
			culture_group = celtic
			culture_group = central_germanic
			culture_group = italian_group
			culture_group = south_slavic
		}
	}
}

religion_spread = {
	mult_modifier = {
		factor = 0.1
		religion = celtic_pagan
		NOT = {
			has_alternate_start_setting = {
				setting = religion
				option = full_random
			}
		}
	}
}

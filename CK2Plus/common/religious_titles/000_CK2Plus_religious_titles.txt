	religion = bogomilist
	can_elect_head = yes
	revoke_allowed = no

	grant_limit = 5

	realm_in_name = no
	show_as_title = yes

	opinion_effect = 20
	dignity = 0.9

	monthly_salary = 0.1
	monthly_prestige = 0.5

	allow = {
		religion = bogomilist
		trait = perfecti
		ai = yes
		d_bogomilist = {
			holder_scope = {
				is_within_diplo_range = ROOT
			}
		}
	}

		character_event = { id = GoPagan.155 }
		set_character_flag = is_cardinal
	}

	lose_effect = {
		clr_character_flag = is_cardinal
	}

	message = no
}
